with open("starbucks_nutrition_17_2019.csv",encoding="utf-16-le") as file:
    lines = file.readlines()
    cats,to_del = [],[]
    for i,line in enumerate(lines):
        if line[0] == ',':
            cats.append(lines[i-1].strip())
            to_del.extend((i-1,i))
    to_del.append(i)
columns = lines[to_del[1]].strip().split(', ')[1:]

lines = [x.strip().split(', ') for i,x in enumerate(lines) if i not in to_del[:-1]]
df = pd.DataFrame(lines)
to_correct = df[df[6].notnull()].index[0]
df.iloc[to_correct,0] += ", {}".format(df.iloc[to_correct,1])
df.iloc[to_correct,1:6] = df.iloc[to_correct,2:].values
df = df.drop(6,1)
index = df[0].values
df = df.drop(0,1)
df.index = index
df.columns = columns
#cats = dict(enumerate(cats))
to_del = np.array(to_del[::2],dtype=np.int8)
repeats = to_del[1:] - to_del[:-1] - 1
repeats[:-1] -= 1
df["meal"] = np.repeat(cats,repeats)
